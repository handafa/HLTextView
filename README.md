### 代码引用

- (void)viewDidLoad {<br />
    [super viewDidLoad];<br />
    _textView.layer.borderWidth = 0.5f;<br />
    _textView.layer.borderColor = [UIColor grayColor].CGColor;<br />
    _textView.hintText = @"谢谢您的使用";<br />
    [_textView useLimit];//启用字数限制<br />
}
